1.3. 配置使用数据库

#myapp/models.py

from django.db import models

class Question(models.Model):
    question_text = models.CharField(max_length=200,default='python')
    pub_date = models.DateTimeField(auto_now=True,name='上传时间')

#myapp/admin.py

from django.contrib import admin


from .models import Question

admin.site.register(Question)

#setting.py 注册应用

INSTALLED_APPS = [
    ...
    'myapp',        #添加应用
]

...

DATABASES = {
    'default': {
        'ENGINE': 'django.db.backends.mysql',   #使用MySQL数据库
        'NAME': 'djangoDemo',       #数据库名字
        'USER': 'user1',            #用户名
        'PASSWORD': '123',          #密码
        'HOST': '192.168.3.128',    #数据库地址
        'PORT': '3306',             #数据库端口
    }
}
